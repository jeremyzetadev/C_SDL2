# for complete sdl2 download use
sudo apt update<br>
sudo apt install libsdl2-dev libsdl2-2.0-0 build-essential<br><br>

[Install SDL2_image (for loading various image formats like PNG, JPEG, etc.)]<br>
sudo apt install libjpeg-dev libwebp-dev libtiff5-dev libsdl2-image-dev libsdl2-image-2.0-0<br>

[Install SDL2_mixer (for sound and music)]<br>
sudo apt install libmikmod-dev libfishsound1-dev libsmpeg-dev liboggz2-dev libflac-dev libfluidsynth-dev libsdl2-mixer-dev libsdl2-mixer-2.0-0<br>

[Install SDL2_ttf (for TrueType font rendering)]<br>
sudo apt install libfreetype6-dev libsdl2-ttf-dev libsdl2-ttf-2.0-0
